/****** Object:  View [dbo].[vUnbalancedJournalEntries]    Script Date: 11-Nov-25 3:14:50 PM ******/
SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
CREATE VIEW [dbo].[vUnbalancedJournalEntries] AS
SELECT je.Id AS JournalEntryId,
       SUM(jed.Debit) AS TotalDebit,
       SUM(jed.Credit) AS TotalCredit
FROM dbo.JournalEntry je
JOIN dbo.JournalEntryDetail jed ON je.Id = jed.JournalEntryId
GROUP BY je.Id
HAVING SUM(jed.Debit) <> SUM(jed.Credit);
GO

CREATE PROCEDURE [dbo].[sp_GetProductDetails]
    @ProductId INT
AS
BEGIN
    SET NOCOUNT ON;

    -- 1. Get Base Product
    SELECT * FROM [dbo].[Product] WHERE Id = @ProductId;

    -- 2. Get Base Product Images
    SELECT * FROM [dbo].[ProductImage] WHERE ProductId = @ProductId ORDER BY SortOrder;

    -- 3. Get Product Variants
    SELECT * FROM [dbo].[ProductVariant] WHERE ProductId = @ProductId AND IsActive = 1;

    -- Create a temp table of the variant IDs for faster lookups
    SELECT Id INTO #VariantIds FROM [dbo].[ProductVariant] WHERE ProductId = @ProductId AND IsActive = 1;

    -- 4. Get Variant Price/Stock (using the temp table)
    SELECT * FROM [dbo].[VariantPriceStock] WHERE Id IN (SELECT Id FROM #VariantIds);

    -- 5. Get Variant-Specific Images (using the temp table)
    SELECT * FROM [dbo].[VariantImage] WHERE VariantId IN (SELECT Id FROM #VariantIds) ORDER BY DisplayOrder;

    -- 6. Get Product Options (e.g., "Color", "Size")
    -- We join AttributeName to get the name
    SELECT
        pa.AttributeId,
        an.Name AS AttributeName
    FROM [dbo].[ProductAttribute] pa
    JOIN [dbo].[AttributeName] an ON pa.AttributeId = an.Id
    WHERE pa.ProductId = @ProductId
    ORDER BY pa.DisplayOrder;

    -- 7. Get All *Possible* Values for those options (e.g., "Silver", "Black", "512GB")
    SELECT
        av.*
    FROM [dbo].[AttributeValue] av
    WHERE av.AttributeId IN (SELECT AttributeId FROM [dbo].[ProductAttribute] WHERE ProductId = @ProductId)
    ORDER BY av.DisplayOrder;

    -- 8. Get the Variant-to-Value Mappings
    SELECT * FROM [dbo].[VariantAttributeValue] WHERE VariantId IN (SELECT Id FROM #VariantIds);

    -- Drop the temp table
    DROP TABLE #VariantIds;

END
GO

    @DiscountType    NVARCHAR(50),   -- Flat or Percentage

    @DiscountValue   DECIMAL(18,2),

    @MinQuantity     INT = NULL,

    @EffectiveFrom   DATETIME,

    @EffectiveTo     DATETIME = NULL,

    @CreatedBy       NVARCHAR(100) = NULL

)

AS

BEGIN

    SET NOCOUNT ON;
 
    --------------------------------------------------------

    -- 1. Validate Discount Type

    --------------------------------------------------------

    IF (@DiscountType NOT IN ('Flat', 'Percentage'))

    BEGIN

        RAISERROR('Invalid DiscountType. Only Flat or Percentage allowed.', 16, 1);

        RETURN;

    END
 
    --------------------------------------------------------

    -- 2. Check for existing discount for this product

    --------------------------------------------------------

    DECLARE @ExistingId INT, @ExistingFrom DATETIME, @ExistingTo DATETIME;
 
    SELECT TOP 1 

           @ExistingId = Id,

           @ExistingFrom = EffectiveFrom,

           @ExistingTo = EffectiveTo

    FROM ProductDiscount

    WHERE ProductId = @ProductId

      AND IsActive = 1

    ORDER BY EffectiveFrom DESC;
 
    --------------------------------------------------------

    -- 3. If no previous discount → just insert

    --------------------------------------------------------

    IF (@ExistingId IS NULL)

    BEGIN

        INSERT INTO ProductDiscount

        (

            ProductId, DiscountType, DiscountValue, MinQuantity,

            EffectiveFrom, EffectiveTo, IsActive,

            CreatedBy

        )

        VALUES

        (

            @ProductId, @DiscountType, @DiscountValue, @MinQuantity,

            @EffectiveFrom, @EffectiveTo, 1,

            @CreatedBy

        );

        RETURN;

    END
 
 
    --------------------------------------------------------

    -- 4. If last discount has no end date → auto close it (N2 + C1)

    --------------------------------------------------------

    IF (@ExistingTo IS NULL)

    BEGIN

        UPDATE ProductDiscount

        SET EffectiveTo = DATEADD(DAY, -1, @EffectiveFrom),

            UpdatedAt = SYSDATETIME(),

            UpdatedBy = @CreatedBy

        WHERE Id = @ExistingId;

    END

    ELSE

    BEGIN

        --------------------------------------------------------

        -- 5. Block overlapping date ranges (OD1)

        --------------------------------------------------------

        IF NOT (@EffectiveFrom > @ExistingTo)

        BEGIN

            RAISERROR('Discount period overlaps with existing discount.', 16, 1);

            RETURN;

        END

    END
 
    --------------------------------------------------------

    -- 6. Insert new discount

    --------------------------------------------------------

    INSERT INTO ProductDiscount

    (

        ProductId, DiscountType, DiscountValue, MinQuantity,

        EffectiveFrom, EffectiveTo, IsActive,

        CreatedBy

    )

    VALUES

    (

        @ProductId, @DiscountType, @DiscountValue, @MinQuantity,

        @EffectiveFrom, @EffectiveTo, 1,

        @CreatedBy

    );

END

GO

CREATE   PROCEDURE [dbo].[usp_RecordCustomerPaymentJournal]
    @CustomerPaymentId INT
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @Amount DECIMAL(18,2), @CustomerId INT, @Status NVARCHAR(20);

    SELECT @Amount = Amount, @CustomerId = CustomerId, @Status = Status
    FROM dbo.CustomerPayment
    WHERE Id = @CustomerPaymentId;

    IF @Status <> 'Completed' RETURN;

    DECLARE @JournalId INT;
    INSERT INTO dbo.JournalEntry (ReferenceType, ReferenceId, Description)
    VALUES ('CustomerPayment', @CustomerPaymentId, CONCAT('Customer Payment #', @CustomerPaymentId));

    SET @JournalId = SCOPE_IDENTITY();

    -- Debit Cash
    INSERT INTO dbo.JournalEntryDetail (JournalEntryId, AccountId, Debit)
    SELECT @JournalId, a.Id, @Amount FROM dbo.Account a WHERE a.AccountCode = '1000';

    -- Credit A/R
    INSERT INTO dbo.JournalEntryDetail (JournalEntryId, AccountId, Credit)
    SELECT @JournalId, a.Id, @Amount FROM dbo.Account a WHERE a.AccountCode = '1100';
END;

/***************************************
-- Procedure: usp_RecordInventory_OUT
-- Purpose : Record outbound stock and decrement inventory
-- Rules   : Must supply SalesOrderDetailId (price comes from SalesOrderDetail.UnitPrice unless @SellingPrice provided)
-- Note    : AverageCost is NOT changed on OUT
***************************************/
CREATE   PROCEDURE [dbo].[usp_RecordInventory_OUT]
(
    @ProductId INT,
    @SalesOrderDetailId INT,
    @Quantity INT,
    @SellingPrice DECIMAL(18,2) = NULL  -- optional; if NULL, will use SalesOrderDetail.UnitPrice
)
AS
BEGIN
    SET NOCOUNT ON;
    SET XACT_ABORT ON;

    -- basic validation
    IF @SalesOrderDetailId IS NULL
    BEGIN
        RAISERROR('SalesOrderDetailId is required for OUT transactions.',16,1);
        RETURN;
    END

    IF @Quantity <= 0
    BEGIN
        RAISERROR('Quantity must be greater than zero.',16,1);
        RETURN;
    END

    DECLARE @UnitPrice DECIMAL(18,2);

    SELECT @UnitPrice = UnitPrice
    FROM dbo.SalesOrderDetail
    WHERE Id = @SalesOrderDetailId;

    IF @UnitPrice IS NULL
    BEGIN
        RAISERROR('SalesOrderDetail.Id = %d not found or UnitPrice is NULL.',16,1,@SalesOrderDetailId);
        RETURN;
    END

    IF @SellingPrice IS NULL
        SET @SellingPrice = @UnitPrice;

    BEGIN TRANSACTION;
    BEGIN TRY
        DECLARE @CurrentStock INT;

        SELECT @CurrentStock = CurrentStock
        FROM dbo.ProductInventory WITH (UPDLOCK, HOLDLOCK)
        WHERE ProductId = @ProductId;

        IF @CurrentStock IS NULL
        BEGIN
            RAISERROR('No ProductInventory row found for ProductId = %d.',16,1,@ProductId);
            ROLLBACK TRANSACTION;
            RETURN;
        END

        IF @CurrentStock < @Quantity
        BEGIN
            RAISERROR('Insufficient stock for ProductId = %d. Current=%d, Requested=%d',16,1,@ProductId, @CurrentStock, @Quantity);
            ROLLBACK TRANSACTION;
            RETURN;
        END

        -- Insert OUT InventoryTransaction (Price is selling price)
        INSERT INTO dbo.InventoryTransaction (SalesOrderDetailId, ProductId, InOut, Date, Price, Quantity)
        VALUES (@SalesOrderDetailId, @ProductId, N'OUT', SYSUTCDATETIME(), @SellingPrice, @Quantity);

        -- Reduce stock only (AverageCost unchanged)
        UPDATE dbo.ProductInventory
        SET CurrentStock = CurrentStock - @Quantity,
            UpdatedAt = SYSUTCDATETIME()
        WHERE ProductId = @ProductId;

        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        IF XACT_STATE() <> 0
            ROLLBACK TRANSACTION;

        DECLARE @ErrMsg NVARCHAR(4000) = ERROR_MESSAGE();
        RAISERROR('Error in usp_RecordInventory_OUT: %s',16,1,@ErrMsg);
        RETURN;
    END CATCH
END;
GO

CREATE   PROCEDURE [dbo].[usp_RecordPurchaseJournal]
    @PoReceivedId INT
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @Amount DECIMAL(18,2), @VendorId INT;

    SELECT @Amount = BuyingPrice * ReceivedQuantity,
           @VendorId = pr.VendorId
    FROM dbo.PoReceived r
    JOIN dbo.PoRequested pr ON r.PoRequestedId = pr.Id
    WHERE r.Id = @PoReceivedId;

    IF @Amount IS NULL RETURN;

    DECLARE @JournalId INT;
    INSERT INTO dbo.JournalEntry (ReferenceType, ReferenceId, Description)
    VALUES ('PoReceived', @PoReceivedId, CONCAT('Purchase Received #', @PoReceivedId));

    SET @JournalId = SCOPE_IDENTITY();

    -- Debit Inventory
    INSERT INTO dbo.JournalEntryDetail (JournalEntryId, AccountId, Debit)
    SELECT @JournalId, a.Id, @Amount FROM dbo.Account a WHERE a.AccountCode = '1200';

    -- Credit Accounts Payable
    INSERT INTO dbo.JournalEntryDetail (JournalEntryId, AccountId, Credit)
    SELECT @JournalId, a.Id, @Amount FROM dbo.Account a WHERE a.AccountCode = '2000';
END;

        ChangeDate

CREATE TRIGGER [dbo].[TR_Customer_Audit]
ON [dbo].[Customer]
AFTER INSERT, UPDATE, DELETE
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @Operation NVARCHAR(10);

    -- Handle INSERT
    IF EXISTS (SELECT 1 FROM inserted) AND NOT EXISTS (SELECT 1 FROM deleted)
    BEGIN
        SET @Operation = 'INSERT';

        INSERT INTO [dbo].[AuditLog] (TableName, Operation, PrimaryKeyValue, OldValues, NewValues, ChangedBy, ChangeDate)
        SELECT 
            'Customer',
            @Operation,
            i.Id,
            NULL,
            (SELECT i.* FOR JSON PATH, WITHOUT_ARRAY_WRAPPER),
            SUSER_SNAME(),
            GETDATE()
        FROM inserted i;
    END

    -- Handle DELETE
    ELSE IF EXISTS (SELECT 1 FROM deleted) AND NOT EXISTS (SELECT 1 FROM inserted)
    BEGIN
        SET @Operation = 'DELETE';

        INSERT INTO [dbo].[AuditLog] (TableName, Operation, PrimaryKeyValue, OldValues, NewValues, ChangedBy, ChangeDate)
        SELECT 
            'Customer',
            @Operation,
            d.Id,
            (SELECT d.* FOR JSON PATH, WITHOUT_ARRAY_WRAPPER),
            NULL,
            SUSER_SNAME(),
            GETDATE()
        FROM deleted d;
    END

    -- Handle UPDATE
    ELSE IF EXISTS (SELECT 1 FROM inserted) AND EXISTS (SELECT 1 FROM deleted)
    BEGIN
        SET @Operation = 'UPDATE';

        -- Update audit log
        INSERT INTO [dbo].[AuditLog] (TableName, Operation, PrimaryKeyValue, OldValues, NewValues, ChangedBy, ChangeDate)
        SELECT 
            'Customer',
            @Operation,
            i.Id,
            (SELECT d.* FOR JSON PATH, WITHOUT_ARRAY_WRAPPER),
            (SELECT i.* FOR JSON PATH, WITHOUT_ARRAY_WRAPPER),
            SUSER_SNAME(),
            GETDATE()
        FROM inserted i
        INNER JOIN deleted d ON i.Id = d.Id;

        -- Update UpdatedBy and UpdatedAt in Customer table
        UPDATE c
        SET 
            c.UpdatedBy = SUSER_SNAME(),
            c.UpdatedAt = GETDATE()
        FROM [dbo].[Customer] c
        INNER JOIN inserted i ON c.Id = i.Id;
    END
END;
GO
